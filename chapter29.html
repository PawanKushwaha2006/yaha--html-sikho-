Chapter 29: HTML Geolocation API

What is Geolocation?

The HTML Geolocation API allows websites to get the geographical position of a user. It uses GPS, WiFi, or IP address to find latitude and longitude. This is done using JavaScript - HTML itself doesn't have a geolocation tag.

🔒 Privacy Important: Geolocation requires user permission. Browser will show a popup asking "Allow location access?" User must approve. It only works on HTTPS sites, not HTTP.

1. Check if Geolocation is Supported

First check if the browser supports Geolocation API:

<script>
if (navigator.geolocation) {
  console.log("Geolocation is supported");
} else {
  console.log("Geolocation not supported");
}
</script>

2. Get Current Position - Basic Example

Use getCurrentPosition() to get user's location once:

<button onclick="getLocation()">Get My Location</button>
<p id="result"></p>

<script>
function getLocation() {
  const result = document.getElementById("result");

  if (navigator.geolocation) {
    result.innerHTML = "Locating...";

    navigator.geolocation.getCurrentPosition(
      showPosition, // Success callback
      showError // Error callback
    );
  } else {
    result.innerHTML = "Geolocation not supported by browser.";
  }
}

function showPosition(position) {
  const lat = position.coords.latitude;
  const lon = position.coords.longitude;
  document.getElementById("result").innerHTML =
    "Latitude: " + lat + "<br>Longitude: " + lon;
}

function showError(error) {
  const result = document.getElementById("result");
  switch(error.code) {
    case error.PERMISSION_DENIED:
      result.innerHTML = "User denied location request.";
      break;
    case error.POSITION_UNAVAILABLE:
      result.innerHTML = "Location info unavailable.";
      break;
    case error.TIMEOUT:
      result.innerHTML = "Request timed out.";
      break;
    default:
      result.innerHTML = "Unknown error occurred.";
  }
}
</script>

3. Position Object Properties

The position object returned has these properties:

position.coords.latitude // Decimal degrees
position.coords.longitude // Decimal degrees
position.coords.accuracy // Meters
position.coords.altitude // Meters above sea level
position.coords.speed // Meters per second
position.coords.heading // Degrees from North
position.timestamp // Time when position was retrieved

4. Watch Position - Track Movement

Use watchPosition() to continuously track user location:

<button onclick="startTracking()">Start Tracking</button>
<button onclick="stopTracking()">Stop Tracking</button>

<script>
let watchID;

function startTracking() {
  if (navigator.geolocation) {
    watchID = navigator.geolocation.watchPosition(
      showPosition,
      showError
    );
  }
}

function stopTracking() {
  if (navigator.geolocation) {
    navigator.geolocation.clearWatch(watchID);
    alert("Tracking stopped");
  }
}
</script>

5. Geolocation Options

You can pass options as third parameter to improve accuracy:

const options = {
  enableHighAccuracy: true, // Use GPS if available
  timeout: 5000, // Wait 5 seconds max
  maximumAge: 0 // Don't use cached position
};

navigator.geolocation.getCurrentPosition(
  showPosition,
  showError,
  options
);
Options Explained:
enableHighAccuracy: true - More accurate but slower, uses more battery
timeout: 5000 - Stop trying after 5 seconds
maximumAge: 0 - Always get fresh location, not cached

6. Show Location on Google Maps

Use latitude/longitude to open Google Maps:

function showPosition(position) {
  const lat = position.coords.latitude;
  const lon = position.coords.longitude;

  // Open in Google Maps
  const url = `https://www.google.com/maps?q=${lat},${lon}`;
  window.open(url, '_blank');
}

7. Error Handling

Error Code Constant Meaning
1 PERMISSION_DENIED User clicked "Block"
2 POSITION_UNAVAILABLE GPS/WiFi failed
3 TIMEOUT Took too long to get location

8. Real-World Use Cases

  • Maps & Navigation: Show user location on map, get directions
  • Weather Apps: Show weather for user's current city
  • Food Delivery: Find nearby restaurants
  • Ride Sharing: Uber/Ola use this to find your pickup point
  • Store Locator: Find nearest store branch
  • Social Media: Tag location in posts
✅ Best Practices:
1. Always ask permission - never auto-request location on page load
2. Explain why you need location before requesting
3. Provide fallback if user denies - manual city input
4. Use HTTPS - Geolocation blocked on HTTP sites
5. Don't store location without user consent
6. Use enableHighAccuracy: false to save battery
7. Handle all error cases gracefully
8. Show loading state while fetching location

9. Complete Working Example

<!DOCTYPE html>
<html>
<body>
  <h2>Find My Location</h2>
  <button onclick="getLocation()">Get Location</button>
  <p id="output"></p>

  <script>
    function getLocation() {
      const output = document.getElementById("output");

      if (!navigator.geolocation) {
        output.innerHTML = "Geolocation not supported";
        return;
      }

      output.innerHTML = "Locating...";

      navigator.geolocation.getCurrentPosition(
        (position) => {
          const lat = position.coords.latitude.toFixed(6);
          const lon = position.coords.longitude.toFixed(6);
          output.innerHTML = `
            <strong>Success!</strong><br>
            Latitude: ${lat}<br>
            Longitude: ${lon}<br>
            Accuracy: ${position.coords.accuracy} meters
          `;
        },
        (error) => {
          output.innerHTML = "Error: " + error.message;
        },
        {
          enableHighAccuracy: true,
          timeout: 10000,
          maximumAge: 0
        }
      );
    }
  </script>
</body>
</html>
⚠️ Security Note: Geolocation only works on HTTPS or localhost. On HTTP sites, browsers block it for security. Always test on HTTPS or use localhost for development.
